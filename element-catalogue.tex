\section{Graphic Symbols Font}

\graphicsymbol{`\textcolor{red}{`} qQwWUcbdhjIJK \textcolor{blue}{`}`}

\begin{symbolfont}%
	qQwWeErRtTyYuUiIoOpP\\
	aAsSdDfFgGhHjJkKlL\\
	zZxXcCvVbBnNmM%
\end{symbolfont}%

\megasymbol for tables and symbol usage.

\section{Screen Output}

\begin{screenoutput}
	10 INPUT A$
	20 PRINT "YOU TYPED: ";A$
	30 PRINT
	40 GOTO 10
	RUN
	? MEGA 65
	YOU TYPED: MEGA 65
\end{screenoutput}

\begin{screenoutput}
10 OPEN 1,8,0,"$0:*,P,R
20 : IF DS THEN PRINT DS$: GOTO 100
30 GET#1,X$,X$
40 DO
50 : GET#1,X$,X$: IF ST THEN EXIT
60 : GET#1,BL$,BH$
70 : LINE INPUT#1, F$
80 : PRINT LEFT$(F$,18)
90 LOOP
100 CLOSE 1

RUN
\end{screenoutput}
